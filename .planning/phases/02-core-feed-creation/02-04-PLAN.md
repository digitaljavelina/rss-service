       ```typescript
       feedsApiRouter.post('/:id/refresh', async (req, res) => {
         const { id } = req.params;

         try {
           // Find feed by id or slug
           const { data: feed, error: feedError } = await supabase
             .from('feeds')
             .select('*')
             .or(`id.eq.${id},slug.eq.${id}`)
             .single();

           if (feedError || !feed) {
             return res.status(404).json({ error: 'Feed not found' });
           }

           // Parse selectors from JSON
           const selectors = JSON.parse(feed.selectors);

           // Fetch page
           const fetchResult = await fetchPage(feed.url);
           if (!fetchResult.ok || !fetchResult.html) {
             return res.status(400).json({
               error: `Failed to fetch source: ${fetchResult.error}`
             });
           }

           // Extract items
           const extractedItems = extractItems(fetchResult.html, feed.url, selectors);

           // Get existing GUIDs to avoid duplicates
           const { data: existingItems } = await supabase
             .from('items')
             .select('guid')
             .eq('feed_id', feed.id);

           const existingGuids = new Set(existingItems?.map(i => i.guid) || []);

           // Filter to new items only
           const newItems = [];
           for (const item of extractedItems) {
             const guid = createHash('sha256')
               .update(`${feed.id}:${item.title}:${item.link}`)
               .digest('hex')
               .substring(0, 16);

             if (!existingGuids.has(guid)) {
               newItems.push({
                 id: nanoid(),
                 feed_id: feed.id,
                 title: item.title,
                 link: item.link,
                 description: item.description || null,
                 pub_date: item.pubDate?.toISOString() || new Date().toISOString(),
                 guid: guid
               });
             }
           }

           // Insert new items
           if (newItems.length > 0) {
             await supabase.from('items').insert(newItems);
           }

           // Enforce item limit - delete oldest items if over limit
           const { count } = await supabase
             .from('items')
             .select('*', { count: 'exact', head: true })
             .eq('feed_id', feed.id);

           if (count && count > feed.item_limit) {
             const toDelete = count - feed.item_limit;
             const { data: oldItems } = await supabase
               .from('items')
               .select('id')
               .eq('feed_id', feed.id)
               .order('pub_date', { ascending: true })
               .limit(toDelete);

             if (oldItems && oldItems.length > 0) {
               await supabase
                 .from('items')
                 .delete()
                 .in('id', oldItems.map(i => i.id));
             }
           }

           // Update feed's updated_at timestamp
           await supabase
             .from('feeds')
             .update({ updated_at: new Date().toISOString() })
             .eq('id', feed.id);

           // Invalidate cache
           invalidateFeedCache(feed.slug);
           invalidateFeedCache(feed.id);

           res.json({
             success: true,
             newItems: newItems.length,
             totalItems: count ? Math.min(count + newItems.length, feed.item_limit) : newItems.length
           });

         } catch (error) {
           console.error('Refresh error:', error);
           res.status(500).json({
             error: error instanceof Error ? error.message : 'Refresh failed'
           });
         }
       });
       ```

    1. Add GET '/:id/export' endpoint to feedsApiRouter:

       ```typescript
       feedsApiRouter.get('/:id/export', async (req, res) => {
         const { id } = req.params;
         const format = req.query.format === 'atom' ? 'atom1' : 'rss2';

         try {
           // Find feed to get the name for filename
           const { data: feed, error: feedError } = await supabase
             .from('feeds')
             .select('name, slug')
             .or(`id.eq.${id},slug.eq.${id}`)
             .single();

           if (feedError || !feed) {
             return res.status(404).json({ error: 'Feed not found' });
           }

           // Build the feed XML
           const xml = await buildFeed(id, format);
           if (!xml) {
             return res.status(404).json({ error: 'Feed not found' });
           }

           // Generate filename
           const extension = format === 'atom1' ? 'atom.xml' : 'rss.xml';
           const filename = `${feed.slug}-${extension}`;

           // Set headers for file download
           res.setHeader('Content-Type', format === 'atom1'
             ? 'application/atom+xml; charset=utf-8'
             : 'application/rss+xml; charset=utf-8');
           res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
           res.send(xml);

         } catch (error) {
           console.error('Export error:', error);
           res.status(500).json({
             error: error instanceof Error ? error.message : 'Export failed'
           });
         }
       });
       ```

    2. Import buildFeed at the top of the file:
       ```typescript
       import { buildFeed } from '../../services/feed-builder.js';
       ```

    The endpoint:
    - Accepts optional ?format=atom query param (default: rss)
    - Sets Content-Disposition header for file download
    - Uses feed slug for filename (e.g., "my-blog-rss.xml")
