<task type="auto">
  <name>Task 3: Update feeds API to handle refresh interval</name>
  <files>src/routes/api/feeds.ts</files>
  <action>
    Update the feeds API to accept and store refresh interval:

    POST /api/feeds (create):
    - Accept refresh_interval_minutes in request body
    - Calculate next_refresh_at = now + interval (if interval is set)
    - Store both values in database

    PUT /api/feeds/:id (update):
    - Accept refresh_interval_minutes in request body
    - Recalculate next_refresh_at if interval changes
    - Clear next_refresh_at if interval is set to null (manual only)

    GET /api/feeds/:id:
    - Include refresh_interval_minutes in response

    GET /api/feeds:
    - Include refresh_interval_minutes in feed list response
  </action>
  <verify>
    - TypeScript compiles: `npx tsc --noEmit`
    - POST creates feed with refresh_interval_minutes
    - PUT updates refresh_interval_minutes
    - GET returns refresh_interval_minutes
  </verify>
  <done>
    Feeds API handles refresh interval configuration
  </done>
</task>
